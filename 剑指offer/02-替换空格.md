class Solution {
public:
	void replaceSpace(char *str,int length) {
        int spaceCount = 0;
        for (int i = 0; i < length; ++i)
        {
            if (str[i] == ' ')
                ++spaceCount;
        }
        int newLen = length + spaceCount * 2;
        int strIndex = length - 1;
        int newStrIndex = newLen - 1;
        while (strIndex >= 0)
        {
            if (str[strIndex] == ' ')
            {
                str[newStrIndex--] = '0';
                str[newStrIndex--] = '2';
                str[newStrIndex--] = '%';
                --strIndex;
            }
            else
            {
                str[newStrIndex] = str[strIndex];
                --newStrIndex;
                --strIndex;
            }
        }
	}
};
