从右上角开始移动

```cpp
class Solution {
public:
    bool Find(int target, vector<vector<int> > array) {
        if (array.size() == 0 || array[0].size() == 0)
            return false;
        int row = 0, col = array[0].size() - 1;
        while (row <= array.size() - 1 && col >= 0)
        {
            if (target < array[row][col])
                --col;
            else if (target > array[row][col])
                ++row;
            else
                return true;
        }
        return false;
    }
};

```
