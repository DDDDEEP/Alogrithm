```cpp
class Solution
{
public:
    void push(int node) {
        stack1.push(node);
    }

    int pop() {
        // 当 stack2 为空时，将 stack1 全部出栈到 stack2
        if (stack2.empty())
        {
            while (!stack1.empty())
            {
                int val1 = stack1.top();
                stack1.pop();
                stack2.push(val1);
            }
        }
        int res = stack2.top();
        stack2.pop();
        return res;
    }

private:
    stack<int> stack1;
    stack<int> stack2;
};
```
